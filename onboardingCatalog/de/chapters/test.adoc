<form id="dataform">
  <input name="projektname" placeholder="Projektname"><br>
  <input name="kunde" placeholder="Kunde"><br>
  <input name="datum" type="date"><br>
  <button type="button" onclick="exportForm()">Exportieren</button>
  <button type="button" onclick="importForm()">Importieren</button>
</form>

<script>

/* -------- JSON EXPORT -------- */
function exportForm() {
  const form = document.getElementById("dataform");
  const formData = new FormData(form);
  const obj = Object.fromEntries(formData.entries());

  const blob = new Blob([JSON.stringify(obj, null, 2)], {
    type: "application/json"
  });

  const a = document.createElement("a");
  a.href = URL.createObjectURL(blob);
  a.download = "formular.json";
  a.click();
}

/* -------- JSON IMPORT -------- */

function importForm() {
  const fileInput = document.getElementById("importfile");
  const file = fileInput.files[0];
  if (!file) return;

  const reader = new FileReader();
  reader.onload = function(e) {
    const data = JSON.parse(e.target.result);
    fillForm(data);
    alert("Daten erfolgreich geladen!");
  };
  reader.readAsText(file);
}

}

/* -------- Formularfelder befüllen -------- */
function fillForm(data) {
  const form = document.getElementById("dataform");

  for (const [key, value] of Object.entries(data)) {
    const field = form.querySelector(`[name="${key}"]`);
    if (field) {
      field.value = value;
    }
  }
}

</script>
